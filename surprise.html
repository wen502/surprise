import random from js import document, window import time texts = [ '生日快乐！', 'Happy Birthday!', '生辰喜乐', '天天开心呀', '愿你常喜乐', '平安又顺遂', '万事皆胜意', '笑容永灿烂', '好运常相伴', '不止今日悦', '岁岁皆无忧', '快乐每一天' ] def get_random_color(): r = random.randint(180, 255) g = random.randint(180, 255) b = random.randint(180, 255) return f'rgb({r}, {g}, {b})' for i in range(40): div = document.createElement('div') div.className = 'wish-window' div.textContent = random.choice(texts) div.style.backgroundColor = get_random_color() screen_width = window.innerWidth - 220 screen_height = window.innerHeight - 100 div.style.left = f'{random.randint(0, screen_width)}px' div.style.top = f'{random.randint(0, screen_height)}px' document.body.appendChild(div) time.sleep(0.08)
